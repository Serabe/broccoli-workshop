Trees

  • Ni ficheros ni streams
  • Carpetas o directorios (trees)
  • Datos de entrada y de salida

/* Brocfile.js */

var compileSass = require('broccoli-sass');
var filterCoffeeScript = require('broccoli-coffee');
var mergeTrees = require('broccoli-merge-trees');

var sassDir = 'scss';
var coffeeDir = 'coffeescript';

var styles = compileSass([sassDir], 'app.scss', 'app.css');
var scripts = filterCoffeeScript(coffeeDir);

module.exports = mergeTrees([styles, scripts]);

Encadenamiento

  • Plugins encadenables
  • Directorios o "trees" de entrada
  • "Trees" de salida

/* Brocfile.js */

var JSHinter = require('broccoli-jshint');
var esTranspiler = require('broccoli-babel-transpiler');
var concat = require('broccoli-concat');

var es6Dir = 'app';

var tree = new JSHinter(es6Dir);
tree = esTranspiler(tree);
tree = concat(tree, {
  outputFile: 'app.js'
});

module.exports = tree;
